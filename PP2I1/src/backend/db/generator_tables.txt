CREATE TABLE clients (client_id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, first_name VARCHAR(100), last_name VARCHAR(100), email VARCHAR(100),pwd VARCHAR(512), created_at DATETIME DEFAULT CURRENT_TIMESTAMP,picked_up_volume INTEGER DEFAULT 0,recycled_volume INTEGER DEFAULT 0,status INTEGER DEFAULT 0)
CREATE TABLE trucks(truck_id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, numberplate VARCHAR(100), capacity INTEGER)
CREATE TABLE waste_type(waste_type_id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, waste_type_name VARCHAR(100))
CREATE TABLE bins(bin_id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,owner_id REFERENCES clients(client_id),product_id REFERENCES products(product_id),lat REAL, long REAL, used INTEGER DEFAULT 0,last_emptied DATETIME, last_emptied_by REFERENCES trucks(truck_id) DEFAULT NULL,bought_at DATETIME DEFAULT CURRENT_TIMESTAMP, waste_id REFERENCES waste_type(waste_type_id))
CREATE TABLE pickup(truck_id REFERENCES trucks(truck_id), bin_id REFERENCES bins(bin_id), pickup_time DATETIME DEFAULT CURRENT_TIMESTAMP,  picked_up INTEGER)
CREATE TABLE products(product_id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, product_name VARCHAR(100), price REAL, img_url VARCHAR(100),desc TEXT, volume INTEGER, stock INTEGER)
